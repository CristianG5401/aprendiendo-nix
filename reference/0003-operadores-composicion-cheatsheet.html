// — actualizar un attribute set

{ editor = "vim"; shell = "bash"; }
// { editor = "nvim"; }
# { editor = "nvim"; shell = "bash"; }

El lado derecho gana cuando ambos sets contienen el mismo atributo.

Importante: la actualización es superficial; un set anidado repetido se reemplaza completo.

? — comprobar existencia

{ sshPort = 22; } ? sshPort
# true

Devuelve un boolean. No obtiene el valor.

or — seleccionar con fallback

{ sshPort = 22; }.backupTarget or "/srv/backups"
# "/srv/backups"

or pertenece a attribute selection: intenta seleccionar y usa el fallback si el atributo no existe.

++ — concatenar listas

[ "git" "curl" ] ++ [ "jq" ]
# [ "git" "curl" "jq" ]

Patrón reusable

let
  effectiveConfig = defaults // overrides;
in
{
  tools = effectiveConfig.tools ++ extraTools;
  hasBackup = effectiveConfig ? backupTarget;
  backupTarget = effectiveConfig.backupTarget or "/srv/backups";
}

Errores frecuentes

  • // no hace deep merge: un set anidado del lado derecho reemplaza el anterior.
  • ? solo comprueba existencia; no devuelve el atributo.
  • or forma parte de attribute selection: escribe set.attr or fallback.
  • ++ concatena listas; no actualiza attribute sets.
